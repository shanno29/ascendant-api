const controller = require('../controllers/FriendController');
const router = require('express').Router();
 
router.get('/', (req, res, next) => {
    controller.list()
        .then(data => {
            return res.json(data);
        })
        .catch(next);
});
router.post('/', (req, res, next) => {
    controller.make(req.body)
        .then(data => {
            return res.json(data);
        })
        .catch(next);
});
router.get('/:key', (req, res, next) => {
    controller.lookup(req.params.key)
        .then(data => {
            return res.json(data);
        })
        .catch(next);
});
router.delete('/:key', (req, res, next) => {
    controller.remove(req.params.key)
        .then(data => {
            return res.json(data);
        })
        .catch(next);
});
 
module.exports = router;
